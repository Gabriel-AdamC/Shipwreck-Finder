CREATE TABLE wrecks (
id INTEGER PRIMARY KEY,
kraken_id TEXT,
ship_name TEXT NOT NULL,
year_lost TEXT,
date_lost TEXT,
reported_depth TEXT,
y_coord REAL,
x_coord REAL,
registered_port INTEGER,
burial_location TEXT,
location_row_ID INTEGER,
build_id INTEGER,
FOREIGN KEY (registered_port) REFERENCES ports(id),
FOREIGN KEY (location_row_ID) REFERENCES locations(location_row_ID),
FOREIGN KEY (build_id) REFERENCES builds(build_id)
);

CREATE TABLE locations (
location_row_ID INTEGER PRIMARY KEY AUTOINCREMENT,
ocean_id INTEGER NOT NULL,
country_id INTEGER,
district_id INTEGER,
local_id INTEGER,
details TEXT,
reported_coords TEXT,
coord_conf INTEGER,
verified_coords TEXT,
coord_type INTEGER,
FOREIGN KEY (ocean_id) REFERENCES oceans(ocean_id),
FOREIGN KEY (country_id) REFERENCES countries(country_id),
FOREIGN KEY (district_id) REFERENCES districts(district_id),
FOREIGN KEY (local_id) REFERENCES local(local_id),
FOREIGN KEY (coord_conf) REFERENCES confidence(id),
FOREIGN KEY (coord_type) REFERENCES coord_type(id)
);

CREATE TABLE builds (
build_id INTEGER PRIMARY KEY AUTOINCREMENT,
material_id INTEGER,
wood_id INTEGER,
fastening_id INTEGER,
sheathing_id INTEGER,
purpose_id INTEGER,
type_id INTEGER,
propulsion_id INTEGER,
engine_id INTEGER,
ship_docs TEXT,
ship_details TEXT,
FOREIGN KEY (material_id) REFERENCES materials(material_id),
FOREIGN KEY (wood_id) REFERENCES wood_types(wood_id),
FOREIGN KEY (fastening_id) REFERENCES fastening(fastening_id),
FOREIGN KEY (sheathing_id) REFERENCES sheathing(sheathing_id),
FOREIGN KEY (purpose_id) REFERENCES purpose(purpose_id),
FOREIGN KEY (type_id) REFERENCES type(type_id),
FOREIGN KEY (propulsion_id) REFERENCES propulsion(propulsion_id),
FOREIGN KEY (engine_id) REFERENCES engines(engine_id)
);

CREATE TABLE voyage (
id INTEGER PRIMARY KEY,
ship_id INTEGER NOT NULL,

port_from INTEGER,
port_to INTEGER,
cargo TEXT,
FOREIGN KEY (ship_id) REFERENCES wrecks(id),
FOREIGN KEY (trade_route) REFERENCES trade_routes(id),
FOREIGN KEY (port_from) REFERENCES ports(id),
FOREIGN KEY (port_to) REFERENCES ports(id)
);
